## 🔍 Production Operations Toolkit

```bash
#!/bin/bash
# ═══════════════════════════════════════════════════════════════
#         GOVIND'S PRODUCTION KUBERNETES OPERATIONS PLAYBOOK
#               Junior DevOps Engineer | Ezdat Technology
# ═══════════════════════════════════════════════════════════════

# ── CLUSTER HEALTH ──────────────────────────────────────────────
kubectl get nodes -o wide                          # Node status + IPs
kubectl get pods -A                                # All pods all namespaces
kubectl top nodes                                  # CPU/Memory per node
kubectl top pods -A                                # CPU/Memory per pod

# ── APPLICATION DEBUGGING ───────────────────────────────────────
kubectl logs -f <pod-name> --tail=100             # Stream live logs
kubectl logs <pod-name> --previous                # Logs of crashed pod
kubectl describe pod <pod-name>                   # Full pod details
kubectl exec -it <pod-name> -- /bin/sh            # Shell into container

# ── DEPLOYMENT MANAGEMENT ───────────────────────────────────────
kubectl rollout status deployment <name>          # Track rollout progress
kubectl rollout history deployment <name>         # View rollout history
kubectl rollout undo deployment <name>            # Emergency rollback
kubectl scale deployment <name> --replicas=5      # Manual scale

# ── TROUBLESHOOTING ─────────────────────────────────────────────
kubectl describe pod <pod-name>                   # CrashLoopBackOff fix
kubectl get events --sort-by=.lastTimestamp       # Recent cluster events
kubectl get svc -o wide                           # Service endpoints
kubectl get ingress                               # Ingress rules

# ── SECRETS & CONFIG ────────────────────────────────────────────
kubectl get secret <name> -o jsonpath=\
  "{.data.<key>}" | base64 --decode               # Decode secret value
kubectl create secret generic <name> \
  --from-literal=key=value                        # Create secret

# ── DOCKER OPERATIONS ───────────────────────────────────────────
docker build --no-cache -t image:tag .            # Fresh build
docker stats                                      # Live container stats
docker system prune -af                           # Clean all unused
trivy image image:tag                             # Security scan image
```
